// sw.js - Corregido para NO interceptar peticiones a Google const CACHE_NAME = 'avt-crm-v1'; const urlsToCache = [ '/', '/dashboard.html', '/reservas.html', '/clientes.html', '/seguimiento.html', '/emails.html', '/calendario.html', '/clima.html', '/manifest.json' ]; self.addEventListener('install', event => { console.log('✅ Service Worker instalado'); event.waitUntil( caches.open(CACHE_NAME) .then(cache => cache.addAll(urlsToCache)) .catch(err => console.log('Error en caché:', err)) ); self.skipWaiting(); }); self.addEventListener('activate', event => { console.log('✅ Service Worker activado'); event.waitUntil( caches.keys().then(cacheNames => { return Promise.all( cacheNames.filter(name => name !== CACHE_NAME) .map(name => caches.delete(name)) ); }) ); self.clients.claim(); }); // NO interceptar peticiones a Google Sheets (CRÍTICO para que carguen los datos) self.addEventListener('fetch', event => { const url = new URL(event.request.url); // Dejar pasar TODAS las peticiones a Google sin cachear if (url.hostname.includes('google.com') || url.hostname.includes('gstatic.com') || url.hostname.includes('script.google.com')) { return; // El navegador maneja directamente } // Solo cachear recursos locales event.respondWith( caches.match(event.request) .then(response => { if (response) return response; return fetch(event.request); }) .catch(() => { if (event.request.mode === 'navigate') { return caches.match('/dashboard.html'); } return new Response('Offline', { status: 503 }); }) ); });
