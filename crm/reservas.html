async function cargarReservas() { try { document.getElementById('tablaReservas').innerHTML = '⏳ Cargando...'; const r = await fetch(SHEET_CSV_URL, { cache: 'no-store' }); const csv = await r.text(); const lineas = csv.split('\n'); reservasData = []; let actual = null, le = []; for (let i = 1; i < lineas.length; i++) { const linea = lineas[i].trim(); if (!linea) continue; if (linea.startsWith('"RES-') || linea.startsWith('RES-')) { if (actual && le.length > 0) { const u = le[le.length-1]; const p = u.replace(/^"|"$/g,'').split('","'); if (p.length >= 4) { actual.fecha = p[1]||''; actual.personas = p[2]||'0'; actual.formaPago = p[3]||'No especificada'; } le.pop(); if (le.length > 0) actual.excursion += '\n' + le.join('\n'); } if (actual) reservasData.push(actual); const v = []; let a = '', c = false; for (let j = 0; j < linea.length; j++) { const ch = linea[j]; if (ch === '"') c = !c; else if (ch === ',' && !c) { v.push(a.trim()); a = ''; } else a += ch; } v.push(a.trim()); const l = v.map(x => x.replace(/^"|"$/g,'').trim()); actual = { id: l[0]||`R${i}`, nombre: l[2]||'', apellido: l[3]||'', email: l[4]||'', telefono: l[5]||'', pais: l[6]||'', ciudad: l[7]||'', excursion: l[8]||'', fecha: '', personas: '0', formaPago: 'No especificada', busqueda: '', estado: 'confirmado' }; le = []; } else { if (actual) { const x = linea.replace(/^"|"$/g,'').trim(); if (x && !x.startsWith(',') && x.length > 3) le.push(x); } } } if (actual) { if (le.length > 0) { const u = le[le.length-1]; const p = u.replace(/^"|"$/g,'').split('","'); if (p.length >= 4) { actual.fecha = p[1]||''; actual.personas = p[2]||'0'; actual.formaPago = p[3]||'No especificada'; } le.pop(); if (le.length > 0) actual.excursion += '\n' + le.join('\n'); } reservasData.push(actual); } // ✅ ORDENAR DE MÁS RECIENTE A MÁS ANTIGUA reservasData.sort((a, b) => { const numA = parseInt(a.id.replace(/\D/g, '')) || 0; const numB = parseInt(b.id.replace(/\D/g, '')) || 0; return numB - numA; }); const eg = cargarEstadosGuardados(); reservasData.forEach(r => { if (eg[r.id]) r.estado = eg[r.id]; }); renderReservas(reservasData); actualizarContadores(); } catch(e) { document.getElementById('tablaReservas').innerHTML = 'Error'; } }
